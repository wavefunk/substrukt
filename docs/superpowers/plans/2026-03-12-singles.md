# Singles Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add "single" content type support — one-off objects (site settings, about page) with no list view, using the existing collection system with a fixed `_single` entry ID.

**Architecture:** Add a `Kind` enum (`Single`/`Collection`) to `SubstruktMeta`. Singles reuse existing content CRUD with a fixed `_single` entry ID. Web routes redirect to edit form directly. API gets dedicated `/single` endpoints registered before `{entry_id}` wildcards. Collection `POST` endpoint rejects singles.

**Tech Stack:** Rust, Axum, serde, minijinja, htmx

**Spec:** `docs/superpowers/specs/2026-03-12-singles-design.md`

async fn get_single(
    State(state): State<AppState>,
    _token: BearerToken,
    Path(schema_slug): Path<String>,
) -> impl IntoResponse {
    let schema_file = match schema::get_schema(&state.config.schemas_dir(), &schema_slug) {
        Ok(Some(s)) => s,
        Ok(None) => return StatusCode::NOT_FOUND.into_response(),
        Err(e) => {
            return (
                StatusCode::INTERNAL_SERVER_ERROR,
                Json(serde_json::json!({"error": e.to_string()})),
            )
                .into_response();
        }
    };

    match content::get_entry(&state.config.content_dir(), &schema_file, "_single") {
        Ok(Some(entry)) => Json(entry.data).into_response(),
        Ok(None) => StatusCode::NOT_FOUND.into_response(),
        Err(e) => (
            StatusCode::INTERNAL_SERVER_ERROR,
            Json(serde_json::json!({"error": e.to_string()})),
        )
            .into_response(),
    }
}

async fn upsert_single(
    State(state): State<AppState>,
    _token: BearerToken,
    Path(schema_slug): Path<String>,
    Json(data): Json<serde_json::Value>,
) -> impl IntoResponse {
    let schema_file = match schema::get_schema(&state.config.schemas_dir(), &schema_slug) {
        Ok(Some(s)) => s,
        Ok(None) => return StatusCode::NOT_FOUND.into_response(),
        Err(e) => {
            return (
                StatusCode::INTERNAL_SERVER_ERROR,
                Json(serde_json::json!({"error": e.to_string()})),
            )
                .into_response();
        }
    };

    if let Err(errors) = content::validate_content(&schema_file, &data) {
        return (
            StatusCode::BAD_REQUEST,
            Json(serde_json::json!({"errors": errors})),
        )
            .into_response();
    }

    let hashes = uploads::extract_upload_hashes(&data);
    match content::save_entry(
        &state.config.content_dir(),
        &schema_file,
        Some("_single"),
        data,
    ) {
        Ok(_) => {
            crate::cache::reload_entry(
                &state.cache,
                &state.config.content_dir(),
                &schema_file,
                "_single",
            );
            let _ =
                uploads::db_update_references(&state.pool, &schema_slug, "_single", &hashes).await;
            state.audit.log(
                "api",
                "content_update",
                "content",
                &format!("{schema_slug}/_single"),
                None,
            );
            StatusCode::OK.into_response()
        }
        Err(e) => (
            StatusCode::INTERNAL_SERVER_ERROR,
            Json(serde_json::json!({"error": e.to_string()})),
        )
            .into_response(),
    }
}
